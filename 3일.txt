     <!-- li 태그는 ul과 ol만 사용할수있다  li안에는 li를 쓸수없다   -->
  <!-- li 와 dd 안에는 li dd 빼고 전부 사용가능 -->
  
               예시  --↓↓↓↓

   <ul>
        <li>아메리카노
            <!-- li 사용안됨 -->
            <ul>
                <li>아이스아메리카노</li>
                <li>아메리카노</li>
            </ul>
        </li>
        <li>카페라떼</li>
        <li>우유
            <ol>
                <li>커피우유</li>
                <li>딸기우유</li>
                <li>초코우유</li>
                <li>우유</li>
            </ol>
        </li>
        <li>차</li>
        <li>카페모카</li>
    </ul>



     <!-- ul은 li의 부모   -->
   <!-- li는 ul의 자식 -->

   <li>첫째</li>
   <li>둘째</li>

   h3+p+ul   -형제

    h3+ul>li*3^  -캐럿 ul
    h3+ul>li*3^^p  -캐럿 h3

    -예시-
    <h3></h3>
    <ul>
        <li></li>
        <li></li>
        <li></li>
    </ul>
    <p></p>
